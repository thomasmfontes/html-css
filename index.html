const display = document.getElementById('result'); const buttons = document.querySelectorAll('.btn'); let currentInput = ''; let operator = ''; let operand1 = ''; let resetDisplay = false; buttons.forEach(btn => { btn.addEventListener('click', () => handleInput(btn.value)); }); function handleInput(value) { if (!isNaN(value)) { appendNumber(value); } else if (['+', '-', 'x', '/'].includes(value)) { setOperation(value); } else if (value === '=') { calculate(); } else if (value === 'C') { clearDisplay(); } else if (value === '.') { appendDot(); } } function appendNumber(num) { if (resetDisplay) { display.value = ''; resetDisplay = false; } display.value += num; } function appendDot() { if (resetDisplay) { display.value = '0'; resetDisplay = false; } // evita dois pontos decimais no mesmo número const lastPart = display.value.split(/[\+\-x\/]/).pop(); if (!lastPart.includes('.')) { display.value += '.'; } } function setOperation(op) { if (display.value === '') return; if (operator) return; // evita repetir operação operand1 = display.value; operator = op; display.value += ` ${op} `; resetDisplay = false; } function calculate() { if (!operator || !display.value.includes(operator)) return; const parts = display.value.split(` ${operator} `); const num1 = parseFloat(parts[0]); const num2 = parseFloat(parts[1]); let result = 0; switch (operator) { case '+': result = num1 + num2; break; case '-': result = num1 - num2; break; case 'x': result = num1 * num2; break; case '/': result = num2 !== 0 ? num1 / num2 : 'Erro'; break; } display.value = result === 'Erro' ? result : (result % 1 === 0 ? result : result.toFixed(2)); // Reset estado operator = ''; operand1 = ''; resetDisplay = true; } function clearDisplay() { display.value = ''; currentInput = ''; operator = ''; operand1 = ''; resetDisplay = false; } document.addEventListener('keydown', (e) => { const key = e.key; if (!isNaN(key)) appendNumber(key); else if (['+', '-', '*', '/'].includes(key)) setOperation(key === '*' ? 'x' : key); else if (key === 'Enter' || key === '=') calculate(); else if (key === '.') appendDot(); else if (key === 'Backspace') display.value = display.value.slice(0, -1); else if (key.toLowerCase() === 'c') clearDisplay(); });
